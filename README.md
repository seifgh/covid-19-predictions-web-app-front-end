```
$ npm --version
6.13.4

```

```
$ node --version
v12.16.1


```

### Installing

- Install dependencies
```json
"dependencies": {
    "@testing-library/jest-dom": "^4.2.4",
    "@testing-library/react": "^9.5.0",
    "@testing-library/user-event": "^7.2.1",
    "axios": "^0.19.2",
    "chart.js": "^2.9.3",
    "history": "^4.10.1",
    "react": "^16.13.1",
    "react-dom": "^16.13.1",
    "react-ga": "^2.7.0",
    "react-helmet": "^5.2.1",
    "react-router-dom": "^5.1.2",
    "react-scripts": "3.4.1",
  },
```

```terminal
$ npm install
```
---
### Running
```terminal
$ npm start //for starting reactjs
$ npm run electron-start //for starting electronjs
```
